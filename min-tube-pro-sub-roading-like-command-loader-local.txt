<script>
(function(){
  var BUILD_DONE_MS = 5000;
  var PROGRESS_MS = 6000;
  var startTime = performance.now();
  var logEl = document.getElementById('log');
  var fillEl = document.getElementById('fill');
  var pctEl = document.getElementById('percent');
  var finalInlineContainer = document.getElementById('finalInlineContainer');

  var dirTreeLines = [
"public",
"app",
"3d-img.html",
"ai.html",
"compiler.html",
"dl.html",
"games.html",
"html-tube.html",
"labo5.html",
"link.html",
"notsecret.html",
"paste.html",
"proxy.html",
"samunedl.html",
"think.html",
"yt.html",
"chat",
"chat.html",
"game",
"battlestars.html",
"drive.html",
"mine.html",
"paper.html",
"snow.html",
"snowride.html",
"js",
"firebase-init.js",
"node-un-server.js",
"oasobi",
"HD",
"mp4quality.html",
"gisou-page.html",
"glfx.js",
"night.html",
"staticproxy",
"helios.html",
"min2-p.html",
"sub",
"robots.html",
"sub-README.md",
"tools",
"about.html",
"api.html",
"apps.html",
"oasobi.html",
"set.html",
"txt",
"img.html",
"min-cat.txt",
"error.html",
"home.html",
"views",
"robots.ejs",
".glitch-assets",
"Procfile",
"index.js",
"package.json",
"railway.json",
"render.yaml",
"shrinkwrap.yaml",
"vercel.json"
  ];
  function buildTreeString(lines){
    var out = [];
    out.push('project/');
    out.push('  ├─ public/');
    var i;
    var grouped = {
      'app': [],
      'chat': [],
      'game': [],
      'js': [],
      'oasobi': [],
      'sub': [],
      'tools': [],
      'views': [],
      'rootFiles': []
    };
    for(i=0;i<lines.length;i++){
      var v = lines[i];
      if(v === 'app' || v === 'chat' || v === 'game' || v === 'js' || v === 'oasobi' || v === 'sub' || v === 'tools' || v === 'views'){
        grouped[v].push(v + '/');
      } else if(v === 'public' || v === 'HD' || v === 'staticproxy') {
      } else {
        if(v.indexOf('.') !== -1) grouped.rootFiles.push(v);
        else grouped.app.push(v);
      }
    }
    out.push('  ├─ app/');
    var appChildren = ['3d-img.html','ai.html','compiler.html','dl.html','games.html','html-tube.html','labo5.html','link.html','notsecret.html','paste.html','proxy.html','samunedl.html','think.html','yt.html'];
    for(i=0;i<appChildren.length;i++) out.push('  │   ├─ ' + appChildren[i]);
    out.push('  ├─ chat/');
    out.push('  │   ├─ chat.html');
    out.push('  ├─ game/');
    var gameChildren = ['battlestars.html','drive.html','mine.html','paper.html','snow.html','snowride.html'];
    for(i=0;i<gameChildren.length;i++) out.push('  │   ├─ ' + gameChildren[i]);
    out.push('  ├─ js/');
    out.push('  │   ├─ firebase-init.js');
    out.push('  │   ├─ node-un-server.js');
    out.push('  ├─ oasobi/');
    out.push('  │   ├─ oasobi.html');
    out.push('  ├─ HD/');
    out.push('  ├─ staticproxy/');
    out.push('  ├─ sub/');
    out.push('  │   ├─ robots.html');
    out.push('  │   ├─ sub-README.md');
    out.push('  ├─ tools/');
    out.push('  │   ├─ about.html');
    out.push('  │   ├─ api.html');
    out.push('  │   ├─ apps.html');
    out.push('  │   ├─ set.html');
    out.push('  ├─ txt/');
    out.push('  │   ├─ img.html');
    out.push('  │   ├─ min-cat.txt');
    out.push('  ├─ views/');
    out.push('  │   ├─ robots.ejs');
    out.push('  ├─ .glitch-assets');
    out.push('  ├─ Procfile');
    out.push('  ├─ index.js');
    out.push('  ├─ package.json');
    out.push('  ├─ railway.json');
    out.push('  ├─ render.yaml');
    out.push('  ├─ shrinkwrap.yaml');
    out.push('  └─ vercel.json');
    return out.join('\n');
  }

  var treeString = buildTreeString(dirTreeLines);

  function randInt(a,b){ return Math.floor(Math.random()*(b-a+1))+a; }
  function randFile(){ var names=['vendor','app','runtime','styles','chunk','entry','polyfills']; return names[randInt(0,names.length-1)] + '.' + Math.random().toString(36).slice(2,10) + '.js'; }
  function randomVersion(){ return randInt(0,3) + '.' + randInt(0,9) + '.' + randInt(0,99); }

  var baseMessages = [
    'loading workspace configuration and environment variables',
    'resolving dependency graph for 312 modules',
    'found cached transform for node_modules/react',
    'analyzing imports and exports across entrypoints',
    'detecting circular dependencies: 2 candidates',
    'applying safe resolution for circular references',
    'preparing incremental build hints',
    'reading tsconfig and babel config',
    'initializing plugin pipeline: eslint, babel, terser, cssnano',
    'compiling typescript sources with isolatedModules',
    'transforming JSX and applying runtime helpers',
    'compiling server-side render bundle',
    'compiling client-side entrypoints',
    'generating module graph and chunk groups',
    'optimizing chunk split points',
    'performing tree-shaking across all modules',
    'marking 124 exports as removable',
    'inlining small modules to reduce requests',
    'extracting CSS and computing critical styles',
    'minifying CSS with safe optimizations',
    'running terser minifier with keep_fnames exceptions',
    'preserving stack-friendly names for core modules',
    'generating source maps (separate files)',
    'serializing asset manifest with gzip and brotli sizes',
    'computing content-hashes for cache busting',
    'writing assets to disk: dist/static/',
    'verifying integrity checksums',
    'preparing service-worker precache list',
    'computing runtime chunk map',
    'embedding runtime flags and environment markers',
    'verifying third-party licenses and generating NOTICE',
    'running post-build audit for large dependencies',
    'applying dedupe strategy for shared modules',
    'uploading artifacts to remote cache',
    'deduplicating identical blobs',
    'finalizing sourcemaps and rewriting sources',
    'sealing artifact bundle for deployment',
    'cleaning temporary build files',
    'optimizing images and compressing assets',
    'preparing deployment package'
  ];

  var messages = [];
  for(var i=0;i<baseMessages.length;i++) messages.push(baseMessages[i]);
  for(var i=0;i<120;i++){
    var idx = randInt(0,baseMessages.length-1);
    messages.splice(randInt(0,messages.length),0, baseMessages[idx] + ' (task ' + randInt(1,999) + ')');
  }
  // insert tree and ascii at strategic points
  messages.splice(4,0, treeString);
  messages.splice(12,0, [
    '   ____  _   _  ____  _  _  ____  ',
    '  (  _ \\( )_( )(  _ \\( \\/ )(  _ \\ ',
    '   ) _ < ) _ (  ) _ < )  /  ) _ < ',
    '  (____/(_) (_)(____/(_/\\_)(____/ ',
    '       R E N D E R   C I   B U I L D'
  ].join('\n'));

  // ensure final success message exists
  messages.push('finalizing chunk graph and runtime manifest');
  messages.push('✔ BUILD SUCCESS');

  var schedule = [];
  var totalEntries = messages.length;
  var acc = 0;
  var minGap = 20;
  var maxGap = 200;
  for(var i=0;i<totalEntries;i++){
    var remaining = BUILD_DONE_MS - acc;
    var remainingEntries = totalEntries - i;
    var maxNext = Math.max(minGap, Math.floor(remaining - (remainingEntries-1)*minGap));
    var delta = (i === totalEntries-1) ? remaining : Math.min(maxGap, Math.max(minGap, Math.floor(Math.random() * maxNext)));
    acc += delta;
    schedule.push({time: acc, text: messages[i]});
  }

  emitLine({text:'agent: starting build agent v' + randomVersion()});

  schedule.forEach(function(s){
    setTimeout(function(){ emitLine({text: s.text}); }, s.time);
  });

  setTimeout(function(){
    emitLine({text:'✔ BUILD SUCCESS'});
    var finalLine = document.createElement('div');
    finalLine.className = 'line';
    var ts = document.createElement('span'); ts.className='ts'; ts.textContent = formatTime(BUILD_DONE_MS);
    var meta = document.createElement('span'); meta.className='meta'; meta.textContent = '[info]';
    var msg = document.createElement('span'); msg.className='msg ok';
    msg.textContent = 'Rendered in 5.00s — production bundle ready 🎉';
    finalLine.appendChild(ts); finalLine.appendChild(meta); finalLine.appendChild(msg);
    finalInlineContainer.appendChild(finalLine);
    finalInlineContainer.setAttribute('aria-hidden','false');
  }, BUILD_DONE_MS);

  var phases = [
    {end:0.12, target:6},
    {end:0.30, target:24},
    {end:0.55, target:50},
    {end:0.80, target:76},
    {end:0.95, target:90},
    {end:1.00, target:100}
  ];
  var progStart = startTime;
  var progDuration = PROGRESS_MS;
  var currentPct = 0;
  var stalls = [];
  for(var s=0;s<3;s++) stalls.push( Math.floor(Math.random() * progDuration * 0.8) + 150 );

  function getPhaseTarget(norm){
    for(var i=0;i<phases.length;i++){
      if(norm <= phases[i].end) return phases[i].target;
    }
    return 100;
  }
  function inStall(elapsed){
    for(var i=0;i<stalls.length;i++){
      var d = Math.abs(elapsed - stalls[i]);
      if(d < 260) return true;
    }
    return false;
  }
  function rafTick(){
    var now = performance.now();
    var elapsed = Math.min(progDuration, now - progStart);
    var norm = elapsed / progDuration;
    var baseTarget = getPhaseTarget(norm);
    if(inStall(elapsed)){
      baseTarget = Math.max((baseTarget - 6), currentPct);
    } else {
      if(Math.random() < 0.03) baseTarget = Math.min(100, baseTarget + randInt(1,6));
    }
    var lerpFactor = 0.06 + (Math.random() * 0.06);
    currentPct = currentPct + (baseTarget - currentPct) * lerpFactor;
    currentPct += (Math.random() - 0.5) * 0.08;
    if(currentPct < 0) currentPct = 0;
    if(currentPct > 100) currentPct = 100;
    if(elapsed >= progDuration - 8) currentPct = 100;
    var display = Math.round(currentPct);
    fillEl.style.width = display + '%';
    fillEl.setAttribute('aria-valuenow', String(display));
    pctEl.textContent = display + '%';
    if(elapsed < progDuration) requestAnimationFrame(rafTick);
    else {
      fillEl.style.width = '100%';
      fillEl.setAttribute('aria-valuenow','100');
      pctEl.textContent = '100%';
    }
  }
  requestAnimationFrame(rafTick);

  function emitLine(obj){
    var div = document.createElement('div');
    div.className = 'line';
    var spanTs = document.createElement('span');
    spanTs.className = 'ts';
    var now = performance.now();
    var rel = Math.max(0, Math.min(BUILD_DONE_MS, now - startTime));
    spanTs.textContent = formatTime(rel);
    var spanMeta = document.createElement('span');
    spanMeta.className = 'meta';
    spanMeta.textContent = '[info]';
    var spanMsg = document.createElement('span');
    spanMsg.className = 'msg';
    spanMsg.textContent = obj.text;
    if(typeof obj.text === 'string' && obj.text.indexOf('\n') !== -1){
      if(obj.text.indexOf('R E N D E R') !== -1) spanMsg.className = 'msg ascii';
      else spanMsg.className = 'msg tree';
    }
    if(obj.text.indexOf('WARN') !== -1 || obj.text.indexOf('large asset') !== -1) spanMsg.className = 'msg warn';
    if(obj.text.indexOf('ERROR') !== -1) spanMsg.className = 'msg err';
    div.appendChild(spanTs);
    div.appendChild(spanMeta);
    div.appendChild(spanMsg);
    logEl.appendChild(div);
    logEl.scrollTop = logEl.scrollHeight;
  }

  function formatTime(ms){ return '[' + (ms/1000).toFixed(3) + 's]'; }

  logEl.addEventListener('keydown', function(e){
    if(e.key === 'PageDown') logEl.scrollTop += logEl.clientHeight;
    if(e.key === 'PageUp') logEl.scrollTop -= logEl.clientHeight;
  });
})();
</script>
