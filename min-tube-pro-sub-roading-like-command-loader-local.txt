<script>
function toggleOtherLogin() {
  const options = document.getElementById("other-login-options");
  if (options.classList.contains("show")) {
    options.classList.remove("show");
    setTimeout(() => { options.style.display = "none"; }, 300);
  } else {
    options.style.display = "block";
    setTimeout(() => { options.classList.add("show"); }, 10);
  }
}

function executeRedirect(e) {
  if (e) e.preventDefault();
  document.cookie = "humanVerified=true; path=/";
  window.location.href = "/";
}

document.addEventListener("DOMContentLoaded", () => {
  const form = document.querySelector("form");
  if (form) {
    form.addEventListener("submit", executeRedirect);
  }

  const allActionButtons = document.querySelectorAll(".other-login-options button");
  allActionButtons.forEach(btn => {
    btn.addEventListener("click", executeRedirect);
  });
});
</script>
