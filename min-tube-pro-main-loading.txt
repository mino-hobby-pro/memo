<script>
  (function(){
    const DURATION = 5000;
    const start = performance.now();
    const barFill = document.getElementById('barFill');
    const percentText = document.getElementById('percentText');
    const msgText = document.getElementById('msgText');
    const progressRole = document.querySelector('[role="progressbar"]');
    const dotRow = document.getElementById('dotRow');

    const messages = [
      "サーバーに接続中です。",
      "APIを読み込んでいます。",
      "接続先を調整中です。",
      "オンラインのインスタンスを取得中。",
      "ボットではないことを確認中です。"
    ];
    const thresholds = [0.0, 0.18, 0.38, 0.65, 0.85];

    function easeOutCubic(t){ return 1 - Math.pow(1 - t, 3); }

    function getMessageFor(t){
      for(let i = thresholds.length - 1; i >= 0; i--){
        if(t >= thresholds[i]) return messages[i];
      }
      return messages[0];
    }

    function tick(now){
      const elapsed = now - start;
      let t = Math.min(1, elapsed / DURATION);
      const eased = easeOutCubic(t);
      const pct = Math.round(eased * 100);
      barFill.style.width = pct + '%';
      percentText.textContent = pct + '%';
      progressRole.setAttribute('aria-valuenow', pct);

      const msg = getMessageFor(t);
      msgText.textContent = msg;

      animateDots(eased);

      if (t < 1) {
        requestAnimationFrame(tick);
      } else {
        msgText.textContent = "準備完了";
        percentText.textContent = "100%";
        barFill.style.boxShadow = '0 0 28px rgba(31,182,164,0.22), inset 0 -8px 18px rgba(0,0,0,0.45)';
        const logo = document.querySelector('.logo');
        if(logo && !window.matchMedia('(prefers-reduced-motion: reduce)').matches){
          logo.animate([
            { transform: 'translateY(0)' },
            { transform: 'translateY(-6px)' },
            { transform: 'translateY(0)' }
          ], { duration: 600, easing: 'ease-out' });
        }
      }
    }

    function animateDots(progress){
      const pixels = dotRow.children;
      for(let i=0;i<pixels.length;i++){
        const phase = (progress * 5 + i * 0.3) % 1;
        const glow = Math.max(0.15, Math.sin(phase * Math.PI) * 0.9);
        pixels[i].style.boxShadow = `0 ${3 + Math.round(glow*2)}px ${Math.round(glow*2)}px rgba(0,0,0,0.6)`;
        pixels[i].style.opacity = (0.6 + glow*0.4).toFixed(2);
      }
    }

    requestAnimationFrame(tick);
  })();
</script>
